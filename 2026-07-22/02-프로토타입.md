### 코드 스켈레톤 (AI 리포트 생성 — 핵심 로직)

```javascript
// app/api/generate-report/route.js
import Anthropic from '@anthropic-ai/sdk';

const anthropic = new Anthropic();

export async function POST(req) {
  const { studentName, lessonDate, memo } = await req.json();

  const msg = await anthropic.messages.create({
    model: 'claude-haiku-4-5-20251001',
    max_tokens: 400,
    messages: [{
      role: 'user',
      content: `아래 수업 메모를 바탕으로 학부모에게 보낼 짧고 정중한 수업 리포트(2문단, 200자 이내)를 한국어로 작성해줘.

학생: ${studentName}
수업일: ${lessonDate}
수업 메모: ${memo}

리포트는 "안녕하세요"로 시작하지 말고, 수업 내용 요약과 다음 과제로 구성해줘.`
    }]
  });

  // DB에 저장
  await supabase.from('lessons').insert({
    student_id: req.headers.get('student-id'),
    date: lessonDate,
    memo,
    report_text: msg.content[0].text,
  });

  return Response.json({ report: msg.content[0].text });
}
```
