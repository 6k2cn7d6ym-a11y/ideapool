```typescript
// app/api/appointments/route.ts
export async function POST(req: Request) {
  const { customerId, datetime, service } = await req.json()
  const appt = await db.appointment.create({
    data: { customerId, datetime, service, status: 'booked' }
  })
  await scheduleReminders(appt.id, new Date(datetime))
  return Response.json(appt)
}

async function scheduleReminders(apptId: string, dt: Date) {
  const offsets = [24 * 60 * 60 * 1000, 60 * 60 * 1000] // 1일 전, 1시간 전
  for (const offset of offsets) {
    const delay = dt.getTime() - offset - Date.now()
    if (delay > 0) {
      await reminderQueue.add('send', { apptId }, { delay })
    }
  }
}

// workers/reminder.ts
reminderQueue.process('send', async (job) => {
  const appt = await db.appointment.findUnique({
    where: { id: job.data.apptId },
    include: { customer: true, business: true }
  })
  if (!appt || appt.status !== 'booked') return

  // 카카오톡 링크: 별도 비즈니스 채널 없이 공유 가능
  const kakaoLink = buildKakaoShareLink(appt)
  await sendKakaoLink(appt.customer.phone, kakaoLink)
})

function buildKakaoShareLink(appt: Appointment & { business: Business }) {
  const msg = `[${appt.business.name}] 예약 안내\n${format(appt.datetime, 'M월 d일 HH:mm')}\n서비스: ${appt.service}`
  return `https://sharer.kakao.com/talk/friends/picker/link?text=${encodeURIComponent(msg)}`
}
```

```typescript
// lib/tax-calculator.ts
const TAX_BRACKETS_2024 = [
  { limit: 14_000_000,  rate: 0.06, deduction: 0 },
  { limit: 50_000_000,  rate: 0.15, deduction: 1_260_000 },
  { limit: 88_000_000,  rate: 0.24, deduction: 5_760_000 },
  { limit: 150_000_000, rate: 0.35, deduction: 15_440_000 },
  { limit: 300_000_000, rate: 0.38, deduction: 19_940_000 },
  { limit: 500_000_000, rate: 0.40, deduction: 25_940_000 },
  { limit: Infinity,    rate: 0.42, deduction: 35_940_000 },
]

export function calculateTax(totalIncome: number, deductibleExpenses: number): TaxSummary {
  const basicDeduction = 1_500_000  // 기본공제 (근로소득 외)
  const netIncome = Math.max(0, totalIncome - deductibleExpenses - basicDeduction)
  const bracket = TAX_BRACKETS_2024.find(b => netIncome <= b.limit)!
  const estimatedTax = Math.max(0, Math.floor(netIncome * bracket.rate - bracket.deduction))
  return {
    totalIncome,
    netIncome,
    estimatedTax,
    effectiveRate: totalIncome > 0 ? estimatedTax / totalIncome : 0,
  }
}

// app/api/tax/summary/route.ts
export async function GET(req: Request) {
  const { searchParams } = new URL(req.url)
  const year = parseInt(searchParams.get('year') ?? String(new Date().getFullYear()))

  const [incomes, expenses] = await Promise.all([
    db.income.findMany({ where: { userId: getUser(req), year } }),
    db.expense.findMany({ where: { userId: getUser(req), year, isDeductible: true } }),
  ])

  const totalIncome = incomes.reduce((s, i) => s + i.amount, 0)
  const totalWithheld = incomes.reduce((s, i) => s + (i.withholdingAmount ?? 0), 0)
  const totalDeductible = expenses.reduce((s, e) => s + e.amount, 0)
  const { estimatedTax } = calculateTax(totalIncome, totalDeductible)

  return Response.json({
    totalIncome, totalWithheld, totalDeductible,
    estimatedTax,
    refundOrPayable: totalWithheld - estimatedTax,  // 양수 = 환급, 음수 = 추가납부
  })
}
```

### [4] 예약관리 — 명세/코드 갭
- 본문 "자동 리마인더"와 코드 스켈레톤(`sharer.kakao.com`)이 어긋남. sharer 링크는 업주가 클릭 후 대상 선택해야 전송되는 반자동. 진짜 자동 발송은 카카오 알림톡(사업자 채널 심사 필요) 또는 SMS 필수.
- "된다"의 기준(32행) 을 반자동 UX로 재정의하거나, 알림톡/SMS 전환 시 개발 기간·비용 재산정 필요.
- 큐 인프라: BullMQ+Redis는 MVP 오버스펙. Supabase(PostgreSQL) 이미 사용 중 → pg-boss 또는 cron 폴링으로 시작 권장.

### [19] 회비관리 — 카카오 연동 방식 불일치
- 코드(261행)의 `kakaotalk://msg/send?text=` URL scheme은 웹 브라우저에서 불안정(OS·브라우저별 편차). [4]는 `sharer.kakao.com`을 쓰면서 [19]는 스킴을 씀 — 통일 필요.
- 권장: Kakao JavaScript SDK(`Kakao.Share.sendDefault`) 또는 sharer 링크로 [4]와 동일화.
- 4주 예상은 낙관. 개인 카카오페이 QR 발급·회원별 등록 UX 흐름 미포함 → +0.5~1주.

### [15] 세금정리 — 타입/로직 부실
- `calculateTax` 반환값이 선언한 `TaxSummary` 타입과 불일치(year·totalWithheld·totalDeductibleExpenses·refundOrPayable 누락, `effectiveRate`는 타입에 없음). 스켈레톤이지만 시그니처 정합성은 맞춰야 함.
- 명세엔 기타소득(3.3% 원천징수)/사업소득 구분 있으나 계산 로직엔 사업소득 종합소득세 브라켓만. 기타소득 분리과세 경로 별도 반영 필요.
- 세무사 검증 필수 조건(300행)은 유지.

### 절차 관련
- 자율 세션 구조상 마이클·클레버가 사실상 동시 실행되어 사전 검토 불가. 지시문의 순차 절차를 강제하려면 파이프라인 구조 개선 필요(planning팀·대표 판단 영역).
