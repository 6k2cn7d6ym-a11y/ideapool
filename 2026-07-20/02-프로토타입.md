// ── 1. 회비 납부 처리
async function recordPayment(groupId, month, userId, amount) {
  await db
    .collection('payments').doc(groupId)
    .collection(month).doc(userId)
    .set({ amount, status: 'paid', paidAt: FieldValue.serverTimestamp() });
}

// ── 2. 미납자 조회
async function getUnpaid(groupId, month) {
  const members = await db
    .collection('members').doc(groupId)
    .collection('users').get();

  const paidSnap = await db
    .collection('payments').doc(groupId)
    .collection(month)
    .where('status', '==', 'paid').get();

  const paidIds = new Set(paidSnap.docs.map(d => d.id));
  return members.docs
    .map(d => ({ id: d.id, ...d.data() }))
    .filter(m => !paidIds.has(m.id));
}

// ── 3. 미납자 FCM 알림
async function sendReminders(groupId, month) {
  const unpaid = await getUnpaid(groupId, month);
  const tokens = unpaid.map(m => m.fcmToken).filter(Boolean);
  if (!tokens.length) return 0;

  await admin.messaging().sendEachForMulticast({
    tokens,
    notification: {
      title: '회비 납부 알림',
      body: `${month} 회비가 아직 미납 상태입니다.`,
    },
  });
  return tokens.length;
}

// ── 4. 정모 RSVP 업데이트
async function updateRSVP(groupId, eventId, userId, response) {
  // response: 'yes' | 'no' | 'maybe'
  await db
    .collection('events').doc(groupId)
    .collection('list').doc(eventId)
    .update({ [`rsvp.${userId}`]: response });
}

// ── 5. 정모 지출 1/N 정산
async function calcSplit(groupId, eventId) {
  const snap = await db
    .collection('events').doc(groupId)
    .collection('list').doc(eventId).get();
  const { rsvp = {}, expenses = [] } = snap.data();

  const attendees = Object.entries(rsvp)
    .filter(([, r]) => r === 'yes')
    .map(([uid]) => uid);

  const total = expenses.reduce((s, e) => s + e.amount, 0);
  const perPerson = attendees.length ? Math.ceil(total / attendees.length) : 0;

  return { total, perPerson, attendees };
}

**MVP 범위 확정 (마이클↔클레버 합의)**

- **포함 (v1)**: 사장님 캘린더, 자체 예약 링크 생성, 수동 예약 입력, 타 채널 예약 수동 복사, 충돌 감지, SMS 확인/취소 알림, 당일 예약 리스트

// ── 1. 예약 충돌 감지 (핵심)
async function checkConflict(
  shopId: string,
  scheduledAt: Date,
  endsAt: Date,
  excludeBookingId?: string
): Promise<boolean> {
  const result = await db.query(`
    SELECT id FROM bookings
    WHERE shop_id = $1
      AND status NOT IN ('cancelled')
      AND id != $2
      AND scheduled_at < $3
      AND ends_at > $4
  `, [shopId, excludeBookingId ?? '', endsAt, scheduledAt]);

  return result.rows.length > 0; // true = 충돌 있음
}

// ── 2. 예약 생성 (자체 링크 고객 → 수동 입력 모두 사용)
async function createBooking(input: {
  shopId: string;
  serviceId: string;
  customerName: string;
  customerPhone: string;
  scheduledAt: Date;
  source: string;
}) {
  const service = await db.query(
    'SELECT duration_min FROM services WHERE id = $1', [input.serviceId]
  );
  const duration = service.rows[0].duration_min;
  const endsAt = new Date(input.scheduledAt.getTime() + duration * 60_000);

  const conflict = await checkConflict(input.shopId, input.scheduledAt, endsAt);
  if (conflict) throw new Error('해당 시간은 이미 예약되어 있습니다.');

  const { rows } = await db.query(`
    INSERT INTO bookings
      (shop_id, service_id, customer_name, customer_phone,
       scheduled_at, ends_at, source)
    VALUES ($1,$2,$3,$4,$5,$6,$7)
    RETURNING id
  `, [input.shopId, input.serviceId, input.customerName,
      input.customerPhone, input.scheduledAt, endsAt, input.source]);

  // SMS 발송
  await sendSMS(
    input.customerPhone,
    `[예약 확인] ${input.scheduledAt.toLocaleString('ko-KR')} 예약이 완료됐습니다.`
  );

  return rows[0].id;
}

// ── 3. 가능 시간 슬롯 조회 (고객 예약 링크용)
async function getAvailableSlots(
  shopId: string,
  serviceId: string,
  date: Date
): Promise<Date[]> {
  const [shopSnap, serviceSnap] = await Promise.all([
    db.query('SELECT hours FROM shops WHERE id=$1', [shopId]),
    db.query('SELECT duration_min FROM services WHERE id=$1', [serviceId]),
  ]);
  const dayKey = date.toLocaleDateString('en-US', { weekday: 'short' }).toLowerCase();
  const hours = shopSnap.rows[0].hours[dayKey]; // e.g. "10:00-19:00"
  if (!hours) return [];

  const [openH, closeH] = hours.split('-').map((t: string) => {
    const [h, m] = t.split(':').map(Number);
    return h * 60 + m;
  });
  const duration = serviceSnap.rows[0].duration_min;

  // 날짜 범위의 기존 예약 조회
  const dayStart = new Date(date); dayStart.setHours(0,0,0,0);
  const dayEnd   = new Date(date); dayEnd.setHours(23,59,59,999);
  const booked = await db.query(`
    SELECT scheduled_at, ends_at FROM bookings
    WHERE shop_id=$1 AND status NOT IN ('cancelled')
      AND scheduled_at BETWEEN $2 AND $3
  `, [shopId, dayStart, dayEnd]);

  const slots: Date[] = [];
  for (let min = openH; min + duration <= closeH; min += 30) {
    const slotStart = new Date(date);
    slotStart.setHours(Math.floor(min/60), min%60, 0, 0);
    const slotEnd = new Date(slotStart.getTime() + duration * 60_000);

    const conflict = booked.rows.some(
      (b: any) => b.scheduled_at < slotEnd && b.ends_at > slotStart
    );
    if (!conflict) slots.push(slotStart);
  }
  return slots;
}

## 마이클↔클레버 토론 기록

### 쟁점 1: [1]+[4] 통합 여부

**마이클 초안 주장:**
통합이 맞다. 회원+회비 코어가 동일하므로 "일정" 탭 하나 추가하면 두 페르소나 커버. 개발 리소스 절반.

**클레버 반박:**
프로덕트 포지셔닝 흐림 리스크 있음. "회비 앱"과 "모임 종합 관리 앱"은 앱스토어 카테고리·마케팅 메시지가 다름. 단, MVP 단계에서 시장 검증 전에 두 개 코드베이스 유지하는 건 낭비. 통합으로 시작, 시장 반응 보고 분리 여부 결정.

**확정:** 통합 ([A] 하나로). [일정] 탭 기본 ON. 추후 반응 보고 분리 검토.

---

### 쟁점 2: [2] MVP에서 다중 채널 자동 연동 없으면 USP가 없는 거 아닌가

**마이클 초안 주장:**
인스타그램 DM API 제약·네이버 파트너 심사 등 외부 변수 크다. MVP는 수동 입력으로 시작, v2에서 자동화.

**클레버 반박:**
수동 입력만이면 우리 앱을 써야 할 이유가 없음. "다중 채널 통합"이 핵심 USP인데, MVP에서 아무것도 안 되면 초기 사용자 유입 근거 부재. 단순 예약 앱 경쟁은 포화 상태.

**마이클 재검토:**
완전 자동 연동은 불가하지만 "채널 출처 태그 + 통합 캘린더 뷰"는 즉시 구현 가능. 사장님 입장에서 "인스타·카카오·네이버 예약이 한 화면에서 보인다" 자체가 가치. 자동화 유무보다 "통합 보기"가 실질 페인 포인트 해소.

**확정:** MVP에 채널 출처 태그 + 통합 캘린더 뷰 포함 (필수). 빠른 수동 입력 UX에 집중. 자동 연동은 v2. 포지셔닝: "모든 채널 예약을 한 화면에" — 자동화가 아닌 통합 뷰가 핵심 가치로 재정의.

---

### 쟁점 3: [B] 인스타그램 연동 v2에서도 가능한가

**클레버 최종 판단:**
Graph API로 DM 내용 직접 파싱은 Meta 정책 위반 가능성 높음. v2에서도 "인스타에서 앱으로 예약 버튼 연결 (링크인바이오)" 형태만 현실적. DM 자동 파싱 경로는 아예 닫힌 것으로 봐야 함. 이 제약은 개발 로드맵에 명시.
